1 'use strict';
  2 
  3  /**
  4   * Class to create numerals convertor (roman <-> arabic)
  5   * */
  6 function NumeralsTranslator() {
  7     /**
  8      * {roman number: arabic number}
  9      * @type {Object}
 10      * */
 11     const romanToArabicDic = {
 12         'I': 1,
 13         'V': 5,
 14         'X': 10,
 15         'L': 50,
 16         'C': 100,
 17         'D': 500,
 18         'M': 1000,
 19     }
 20 
 21     /**
 22      * {arabic number: roman number}
 23      * @type {Object}
 24      * const arabicToRomanDic = {
 25      *         1: 'I',
 26      *         5: 'V',
 27      *         10: 'X',
 28      *         50: 'L',
 29      *         100: 'C',
 30      *         500: 'D',
 31      *         1000: 'M',
 32      * }
 33      * */
 34     const arabicToRomanDic = Object.fromEntries(Object.entries(romanToArabicDic).map(dic => dic.reverse()))
 35 
 36     /**
 37      * roman numbers in arabic numerics
 38      * @type {Array}
 39      * const romanNums = [1000, 500, 100, 50, 10, 5, 1];
 40      * */
 41     const romanNums = Object.keys(arabicToRomanDic).reverse();
 42     /**
 43      * roman symbols repeated 4 times for usage in .replace()
 44      * @type {Array}
 45      * */
 46     const roman4Symbols = ['MMMM', 'DDDD', 'CCCC', 'LLLL', 'XXXX', 'VVVV', 'IIII'];
 47 
 48     /**
 49      * converts arabic number to roman
 50      * @param {number} arabicN arabic number to convert
 51      * @returns {string} roman number
 52      * */
 53     this.translateArabicToRoman = function (arabicN) {
 54         /**
 55          * resulting roman number
 56          * @type {string}
 57          * */
 58         let romanN = '';
 59 
 60         for (let i = 0; arabicN !== 0; ++i)
 61             if (arabicN - romanNums[i] >= 0) {
 62                 arabicN -= romanNums[i];
 63                 romanN += arabicToRomanDic[romanNums[i]];
 64                 --i;
 65             }
 66 
 67         for (let i = 2; i < roman4Symbols.length; i++) {
 68             romanN = romanN.replace(roman4Symbols[i - 1][0] + roman4Symbols[i], roman4Symbols[i][0] + roman4Symbols[i - 2][0])
 69             romanN = romanN.replace(roman4Symbols[i], roman4Symbols[i][0] + roman4Symbols[i - 1][0])
 70         }
 71 
 72         return romanN;
 73     }
 74 
 75     /**
 76      * converts roman number to arabic
 77      * @param {string} romanN roman number to convert
 78      * @returns {number} arabic number
 79      * */
 80     this.translateRomanToArabic = function (romanN) {
 81         romanN = romanN.toUpperCase().split('');
 82 
 83         /**
 84          * resulting arabic number
 85          * @type {number}
 86          * */
 87         let arabicN = 0;
 88 
 89         for (let i = 0; i < romanN.length - 1; i++) {
 90             if (romanToArabicDic[romanN[i]] >= romanToArabicDic[romanN[i + 1]]) {
 91                 arabicN += romanToArabicDic[romanN[i]];
 92             } else
 93                 arabicN -= romanToArabicDic[romanN[i]];
 94         }
 95 
 96         arabicN += romanToArabicDic[romanN[romanN.length - 1]];
 97         return arabicN;
 98     }
 99 }
100 
101 /**
102  * arabic number validation
103  * @param {string} inputSymbols symbols for check
104  * @returns {boolean} whether +inputSymbols is integer
105  * */
106 let isArabic = (inputSymbols) => Number.isInteger(+inputSymbols);
107 
108 /**
109  * roman number validation
110  * @param {string} inputSymbols symbols for check
111  * @returns {boolean} whether inputSymbols is roman number
112  * */
113 let isRoman = function (inputSymbols) {
114     if (typeof inputSymbols !== 'string')
115         return false;
116 
117     /**
118      * to exclude any non-roman numeric symbols
119      * @type {RegExp}
120      * */
121     const romanChars = /^[MDCLXVI]*$/;
122     return romanChars.test(inputSymbols.toUpperCase());
123 }
124 
125 /**
126  * converts number to different numerics (roman <-> arabic)
127  * @param {string} inputSymbols symbols to convert
128  * @returns {string | number} converted roman | arabic number
129  * */
130 let roman = function (inputSymbols) {
131     /**
132      * See {@link NumeralsTranslator}
133      * */
134     let translator = new NumeralsTranslator;
135 
136     if (isArabic(inputSymbols))
137         return (inputSymbols >= 1 && inputSymbols <= 3999) ?
138             translator.translateArabicToRoman(+inputSymbols) : NaN;
139 
140     if (isRoman(inputSymbols))
141         return translator.translateRomanToArabic(inputSymbols);
142 
143     console.log("Error, wrong input format // roman()");
144     return (NaN);
145 }
